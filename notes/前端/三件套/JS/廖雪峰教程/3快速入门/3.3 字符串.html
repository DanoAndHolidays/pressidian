еҰӮжһң'жң¬иә«д№ҹжҳҜдёҖдёӘеӯ—з¬ҰпјҢйӮЈе°ұеҸҜд»Ҙз”Ё""жӢ¬иө·жқҘпјҢжҜ”еҰӮ"I'm OK"еҢ…еҗ«зҡ„еӯ—з¬ҰжҳҜIпјҢ'пјҢmпјҢз©әж јпјҢOпјҢKиҝҷ6дёӘеӯ—з¬ҰгҖӮ

жЁЎжқҝеӯ—з¬ҰдёІ вҖӢ

иҰҒжҠҠеӨҡдёӘеӯ—з¬ҰдёІиҝһжҺҘиө·жқҘпјҢеҸҜд»Ҙз”Ё+еҸ·иҝһжҺҘпјҡ

javascript
let name = 'е°ҸжҳҺ';
let age = 20;
let message = 'дҪ еҘҪ, ' + name + ', дҪ д»Ҡе№ҙ' + age + 'еІҒдәҶ!';
alert(message);

еҰӮжһңжңүеҫҲеӨҡеҸҳйҮҸйңҖиҰҒиҝһжҺҘпјҢз”Ё+еҸ·е°ұжҜ”иҫғйә»зғҰгҖӮES6ж–°еўһдәҶдёҖз§ҚжЁЎжқҝеӯ—з¬ҰдёІпјҢиЎЁзӨәж–№жі•е’ҢдёҠйқўзҡ„еӨҡиЎҢеӯ—з¬ҰдёІдёҖж ·пјҢдҪҶжҳҜе®ғдјҡиҮӘеҠЁжӣҝжҚўеӯ—з¬ҰдёІдёӯзҡ„еҸҳйҮҸпјҡ

javascript
let name = 'е°ҸжҳҺ';
let age = 20;
let message = `дҪ еҘҪ, ${name}, дҪ д»Ҡе№ҙ${age}еІҒдәҶ!`;
alert(message);

ж“ҚдҪңеӯ—з¬ҰдёІ вҖӢ

иҺ·еҸ–еӯ—з¬ҰдёІй•ҝеәҰпјҡ

javascript
let s = 'Hello, world!';
s.length; // 13

иҰҒиҺ·еҸ–еӯ—з¬ҰдёІжҹҗдёӘжҢҮе®ҡдҪҚзҪ®зҡ„еӯ—з¬ҰпјҢдҪҝз”Ёзұ»дјјArrayзҡ„дёӢж Үж“ҚдҪңпјҢзҙўеј•еҸ·д»Һ0ејҖе§Ӣпјҡ

javascript
let s = 'Hello, world!';

s[0]; // 'H'
s[6]; // ' '
s[7]; // 'w'
s[12]; // '!'
s[13]; // undefined и¶…еҮәиҢғеӣҙзҡ„зҙўеј•дёҚдјҡжҠҘй”ҷпјҢдҪҶдёҖеҫӢиҝ”еӣһundefined

йңҖиҰҒзү№еҲ«жіЁж„Ҹзҡ„жҳҜпјҢеӯ—з¬ҰдёІжҳҜдёҚеҸҜеҸҳзҡ„пјҢеҰӮжһңеҜ№еӯ—з¬ҰдёІзҡ„жҹҗдёӘзҙўеј•иөӢеҖјпјҢдёҚдјҡжңүд»»дҪ•й”ҷиҜҜпјҢдҪҶжҳҜпјҢд№ҹжІЎжңүд»»дҪ•ж•Ҳжһңпјҡ

javascript
let s = 'Test';
s[0] = 'X';
console.log(s); // sд»Қз„¶дёә'Test'

JavaScriptдёәеӯ—з¬ҰдёІжҸҗдҫӣдәҶдёҖдәӣеёёз”Ёж–№жі•пјҢжіЁж„ҸпјҢи°ғз”Ёиҝҷдәӣж–№жі•жң¬иә«дёҚдјҡж”№еҸҳеҺҹжңүеӯ—з¬ҰдёІзҡ„еҶ…е®№пјҢиҖҢжҳҜиҝ”еӣһдёҖдёӘж–°еӯ—з¬ҰдёІпјҡ

toUpperCase вҖӢ

toUpperCase()жҠҠдёҖдёӘеӯ—з¬ҰдёІе…ЁйғЁеҸҳдёәеӨ§еҶҷпјҡ

javascript
let s = 'Hello';
s.toUpperCase(); // иҝ”еӣһ'HELLO'

toLowerCase вҖӢ

toLowerCase()жҠҠдёҖдёӘеӯ—з¬ҰдёІе…ЁйғЁеҸҳдёәе°ҸеҶҷпјҡ

javascript
let s = 'Hello';
let lower = s.toLowerCase(); // иҝ”еӣһ'hello'е№¶иөӢеҖјз»ҷеҸҳйҮҸlower
lower; // 'hello'

indexOf вҖӢ

indexOf()дјҡжҗңзҙўжҢҮе®ҡеӯ—з¬ҰдёІеҮәзҺ°зҡ„дҪҚзҪ®пјҡ

javascript
let s = 'hello, world';
s.indexOf('world'); // иҝ”еӣһ7
s.indexOf('World'); // жІЎжңүжүҫеҲ°жҢҮе®ҡзҡ„еӯҗдёІпјҢиҝ”еӣһ-1

substring вҖӢ

substring()иҝ”еӣһжҢҮе®ҡзҙўеј•еҢәй—ҙзҡ„еӯҗдёІпјҡ

javascript
let s = 'hello, world'
s.substring(0, 5); // д»Һзҙўеј•0ејҖе§ӢеҲ°5пјҲдёҚеҢ…жӢ¬5пјүпјҢиҝ”еӣһ'hello'
s.substring(7); // д»Һзҙўеј•7ејҖе§ӢеҲ°з»“жқҹпјҢиҝ”еӣһ'world'
