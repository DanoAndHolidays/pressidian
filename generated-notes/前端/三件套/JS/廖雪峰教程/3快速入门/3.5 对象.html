JavaScriptзҡ„еҜ№иұЎз”ЁдәҺжҸҸиҝ°зҺ°е®һдё–з•Ңдёӯзҡ„жҹҗдёӘеҜ№иұЎгҖӮдҫӢеҰӮпјҢдёәдәҶжҸҸиҝ°вҖңе°ҸжҳҺвҖқиҝҷдёӘж·ҳж°”зҡ„е°ҸжңӢеҸӢпјҢжҲ‘д»¬еҸҜд»Ҙз”ЁиӢҘе№Ій”®еҖјеҜ№жқҘжҸҸиҝ°д»–пјҡ

javascript
let xiaoming = {
    name: 'е°ҸжҳҺ',
    birth: 1990,
    school: 'No.1 Middle School',
    height: 1.70,
    weight: 65,
    score: null
};

JavaScriptз”ЁдёҖдёӘ{...}иЎЁзӨәдёҖдёӘеҜ№иұЎпјҢй”®еҖјеҜ№д»Ҙxxx: xxxеҪўејҸз”іжҳҺпјҢз”Ё,йҡ”ејҖгҖӮжіЁж„ҸпјҢжңҖеҗҺдёҖдёӘй”®еҖјеҜ№дёҚйңҖиҰҒеңЁжң«е°ҫеҠ ,пјҢеҰӮжһңеҠ дәҶпјҢжңүзҡ„жөҸи§ҲеҷЁпјҲеҰӮдҪҺзүҲжң¬зҡ„IEпјүе°ҶжҠҘй”ҷгҖӮ

дёҠиҝ°еҜ№иұЎз”іжҳҺдәҶдёҖдёӘnameеұһжҖ§пјҢеҖјжҳҜ'е°ҸжҳҺ'пјҢbirthеұһжҖ§пјҢеҖјжҳҜ1990пјҢд»ҘеҸҠе…¶д»–дёҖдәӣеұһжҖ§гҖӮжңҖеҗҺпјҢжҠҠиҝҷдёӘеҜ№иұЎиөӢеҖјз»ҷеҸҳйҮҸxiaomingеҗҺпјҢе°ұеҸҜд»ҘйҖҡиҝҮеҸҳйҮҸxiaomingжқҘиҺ·еҸ–е°ҸжҳҺзҡ„еұһжҖ§дәҶпјҡ

javascript
xiaoming.name; // 'е°ҸжҳҺ'
xiaoming.birth; // 1990

xiaohongзҡ„еұһжҖ§еҗҚmiddle-schoolдёҚжҳҜдёҖдёӘжңүж•Ҳзҡ„еҸҳйҮҸпјҢе°ұйңҖиҰҒз”Ё''жӢ¬иө·жқҘгҖӮи®ҝй—®иҝҷдёӘеұһжҖ§д№ҹж— жі•дҪҝз”Ё.ж“ҚдҪңз¬ҰпјҢеҝ…йЎ»з”Ё['xxx']жқҘи®ҝй—®пјҡ

javascript
xiaohong['middle-school']; // 'No.1 Middle School'
xiaohong['name']; // 'е°Ҹзәў'
xiaohong.name; // 'е°Ҹзәў'

д№ҹеҸҜд»Ҙз”Ёxiaohong['name']жқҘи®ҝй—®xiaohongзҡ„nameеұһжҖ§пјҢдёҚиҝҮxiaohong.nameзҡ„еҶҷжі•жӣҙз®ҖжҙҒгҖӮжҲ‘д»¬еңЁзј–еҶҷJavaScriptд»Јз Ғзҡ„ж—¶еҖҷпјҢеұһжҖ§еҗҚе°ҪйҮҸдҪҝз”Ёж ҮеҮҶзҡ„еҸҳйҮҸеҗҚпјҢиҝҷж ·е°ұеҸҜд»ҘзӣҙжҺҘйҖҡиҝҮobject.propзҡ„еҪўејҸи®ҝй—®дёҖдёӘеұһжҖ§дәҶгҖӮ

е®һйҷ…дёҠJavaScriptеҜ№иұЎзҡ„жүҖжңүеұһжҖ§йғҪжҳҜеӯ—з¬ҰдёІпјҢдёҚиҝҮеұһжҖ§еҜ№еә”зҡ„еҖјеҸҜд»ҘжҳҜд»»ж„Ҹж•°жҚ®зұ»еһӢгҖӮ

еҰӮжһңи®ҝй—®дёҖдёӘдёҚеӯҳеңЁзҡ„еұһжҖ§дјҡиҝ”еӣһд»Җд№Ҳе‘ўпјҹJavaScriptи§„е®ҡпјҢи®ҝй—®дёҚеӯҳеңЁзҡ„еұһжҖ§дёҚжҠҘй”ҷпјҢиҖҢжҳҜиҝ”еӣһundefined

з”ұдәҺJavaScriptзҡ„еҜ№иұЎжҳҜеҠЁжҖҒзұ»еһӢпјҢдҪ еҸҜд»ҘиҮӘз”ұең°з»ҷдёҖдёӘеҜ№иұЎж·»еҠ жҲ–еҲ йҷӨеұһжҖ§пјҡ

javascript
let xiaoming = {
    name: 'е°ҸжҳҺ'
};
xiaoming.age; // undefined
xiaoming.age = 18; // ж–°еўһдёҖдёӘageеұһжҖ§
xiaoming.age; // 18
delete xiaoming.age; // еҲ йҷӨageеұһжҖ§
xiaoming.age; // undefined
delete xiaoming['name']; // еҲ йҷӨnameеұһжҖ§
xiaoming.name; // undefined
delete xiaoming.school; // еҲ йҷӨдёҖдёӘдёҚеӯҳеңЁзҡ„schoolеұһжҖ§д№ҹдёҚдјҡжҠҘй”ҷ

еҰӮжһңжҲ‘д»¬иҰҒжЈҖжөӢxiaomingжҳҜеҗҰжӢҘжңүжҹҗдёҖеұһжҖ§пјҢеҸҜд»Ҙз”Ёinж“ҚдҪңз¬Ұпјҡ

javascript
let xiaoming = {
    name: 'е°ҸжҳҺ',
    birth: 1990,
    school: 'No.1 Middle School',
    height: 1.70,
    weight: 65,
    score: null
};
'name' in xiaoming; // true
'grade' in xiaoming; // false

дёҚиҝҮиҰҒе°ҸеҝғпјҢеҰӮжһңinеҲӨж–ӯдёҖдёӘеұһжҖ§еӯҳеңЁпјҢиҝҷдёӘеұһжҖ§дёҚдёҖе®ҡжҳҜxiaomingзҡ„пјҢе®ғеҸҜиғҪжҳҜxiaomingз»§жүҝеҫ—еҲ°зҡ„пјҡ

javascript
'toString' in xiaoming; // true

еӣ дёәtoStringе®ҡд№үеңЁobjectеҜ№иұЎдёӯпјҢиҖҢжүҖжңүеҜ№иұЎжңҖз»ҲйғҪдјҡеңЁеҺҹеһӢй“ҫдёҠжҢҮеҗ‘objectпјҢжүҖд»Ҙxiaomingд№ҹжӢҘжңүtoStringеұһжҖ§гҖӮ

иҰҒеҲӨж–ӯдёҖдёӘеұһжҖ§жҳҜеҗҰжҳҜxiaomingиҮӘиә«жӢҘжңүзҡ„пјҢиҖҢдёҚжҳҜз»§жүҝеҫ—еҲ°зҡ„пјҢеҸҜд»Ҙз”ЁhasOwnProperty()ж–№жі•пјҡ

javascript
let xiaoming = {
    name: 'е°ҸжҳҺ'
};
xiaoming.hasOwnProperty('name'); // true
xiaoming.hasOwnProperty('toString'); // false
